constructor(props) {
        super(props);
        this.state = {
            location: {},
        };
    }
    componentDidMount() {
        this.getIssLocation()
    }
    getIssLocation = () => {
        axios
            .get("https://api.wheretheiss.at/v1/satellites/25544")
            .then(response => {
                this.setState({ location: response.data })
            })
            .catch(error => {
                Alert.alert(error.message)
            })
    }
